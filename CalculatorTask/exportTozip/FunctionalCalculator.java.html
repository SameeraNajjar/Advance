public class FunctionalCalculator {
public static void main(String[]args){
    Calculator add =(a,b)-> a+b;
    Calculator sub =(a,b)-> a-b;
    Calculator multi =(a,b)-> a*b;
    Calculator div =(a, b)-> {
        if(b==0){
            throw new IllegalArgumentException("cannot div by 0");
        }
        return a/b;
    };
    Calculator mod=(a,b)-> a%b;
    double num1 = 10.0;
    double num2 = 3.0;
    System.out.println("Addition: " + add.calculate(num1, num2));
    System.out.println("Subtraction: " + sub.calculate(num1, num2));
    System.out.println("Multiplication: " + multi.calculate(num1, num2));
    try {
        System.out.println("Division: " + div.calculate(num1, num2));
    } catch (IllegalArgumentException e) {
        System.out.println(e.getMessage());
    }
    System.out.println("Modulus: " + mod.calculate(num1, num2));
}
}
